    Widget textButton(
  String text,
  Size viewportSize, {
  void Function()? onTap,
  ButtonStyle? buttonStyle,
  ContainerImageStyling? containerStyling,
}) {
  Widget result = TextButton(
    child: container(text, viewportSize),
    onPressed: onTap ?? () {},
    style: buttonStyle,
  );

  if (containerStyling != null) {
    result = containerStyling.toWidget(
      viewportSize,
      child: result,
    );
  }

  return result;
}
